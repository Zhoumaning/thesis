\begin{equation}
\frac{P(\bar H|n_{ij})}{P(H|n_{ij})}
\propto

    \prod_i \Gamma(n_{\cdot i}+1)

\end{equation}

\subsection{Laskenta}

\begin{equation}

(\ref{eq:bf1}) pehmennettyjen rajojen logaritmina. N‰it‰ termej‰
